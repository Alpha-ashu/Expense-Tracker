      {/* --- Desktop: Modular Dashboard Layout --- */}
      <div className="hidden lg:flex min-h-screen flex-col bg-[#f4f6f8]">
        <div className="flex-1 w-full max-w-7xl mx-auto px-8 py-10 overflow-y-auto">
          {/* Header */}
          <div className="flex justify-between items-center mb-8">
            <div>
              <h1 className="text-3xl font-black text-gray-900 tracking-tight">
                {isLoanExpense ? 'Record Loan / Debt' : isGroupExpense ? 'Group Expense' : isExpense ? 'Add Expense' : 'Add Income'}
              </h1>
              <p className="text-gray-500 mt-2 font-medium">
                {isLoanExpense ? 'Track money borrowed or lent' : isGroupExpense ? 'Split expenses with friends' : 'Fast, smart financial tracking'}
              </p>
            </div>
            <div className="flex gap-3">
              {isExpense && !isLoanExpense && (
                <button type="button" onClick={() => setShowScanner(true)}
                  className="flex items-center gap-2 px-5 py-3 rounded-2xl bg-white border border-gray-200 text-gray-700 font-bold hover:bg-gray-50 shadow-sm transition-all">
                  <Camera size={18} /> Scan Receipt
                </button>
              )}
              <button type="button" onClick={() => setCurrentPage(returnPage)}
                className="flex items-center gap-2 px-5 py-3 rounded-2xl bg-white border border-gray-200 text-gray-700 font-bold hover:bg-gray-50 shadow-sm transition-all">
                Cancel
              </button>
            </div>
          </div>

          <form onSubmit={handleSubmit} className="grid grid-cols-12 gap-8">
            {/* -- MAIN COLUMN (Left) -- */}
            <div className="col-span-12 lg:col-span-8 space-y-6">

              {/* Amount & Mode Card */}
              <div className="bg-white rounded-[2rem] p-8 shadow-sm border border-gray-100">
                <div className="flex justify-between items-start mb-8">
                  <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest">Transaction Details</h2>
                  <div className="flex items-center gap-1 bg-gray-100 rounded-xl p-1">
                    <button type="button" onClick={() => switchType('expense')}
                      className={cn('px-5 py-2 rounded-lg text-sm font-bold transition-all', isExpense ? 'bg-white text-gray-900 shadow-sm' : 'text-gray-500 hover:text-gray-700')}>
                      Expense
                    </button>
                    <button type="button" onClick={() => switchType('income')}
                      className={cn('px-5 py-2 rounded-lg text-sm font-bold transition-all', !isExpense ? 'bg-white text-gray-900 shadow-sm' : 'text-gray-500 hover:text-gray-700')}>
                      Income
                    </button>
                  </div>
                </div>

                <div className={cn('rounded-[2rem] p-8 flex items-center transition-all bg-gradient-to-r', accent.amountCard)}>
                  <span className="text-4xl font-black text-white/80 mr-4">
                    {isExpense ? '-' : '+'}{currency}
                  </span>
                  <input
                    type="number" step="0.01" min="0"
                    value={amountStr}
                    onChange={e => handleAmountChange(e.target.value)}
                    className="flex-1 bg-transparent text-6xl font-black text-white outline-none placeholder:text-white/40 w-full"
                    placeholder="0.00"
                    required
                    autoFocus
                  />
                </div>
              </div>

              {/* Category & Date Card */}
              <div className="bg-white rounded-[2rem] p-8 shadow-sm border border-gray-100">
                <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest mb-6">Category & Date</h2>
                <div className="grid grid-cols-2 gap-6">
                  <div>
                    {isExpense && !isLoanExpense ? (
                      <div className="relative">
                        <Search size={18} className="absolute left-4 top-1/2 -translate-y-1/2 text-gray-400 pointer-events-none" />
                        <input type="text" value={subcategoryQuery}
                          onChange={e => {
                            const value = e.target.value;
                            setSubcategoryQuery(value);
                            if (!value.trim()) setManualExpenseCategory(false);
                            const exactCat = getCategoryForExpenseSubcategory(value, customExpenseSubcategories);
                            if (exactCat) {
                              setManualExpenseCategory(false);
                              setFormData(prev => ({ ...prev, category: exactCat, subcategory: value.trim() }));
                            } else {
                              setFormData(prev => ({ ...prev, subcategory: '' }));
                            }
                          }}
                          className="w-full pl-12 pr-4 py-4 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                          placeholder="Search categories (e.g. Uber, Netflix)..."
                        />
                        {formData.category && (
                          <div className="mt-4 flex items-center justify-between px-4 py-3 bg-gray-50 rounded-2xl border border-gray-100">
                            <div className="flex items-center gap-3">
                               {getCategoryCartoonIcon(resolvedExpenseCategory, 24)}
                               <span className="text-sm font-bold text-gray-900">{resolvedExpenseCategory}</span>
                            </div>
                            <button type="button" onClick={() => setShowCategoryPicker(p => !p)}
                              className="text-xs font-bold text-blue-600 hover:text-blue-700 bg-blue-50 px-3 py-1.5 rounded-lg">
                              Change
                            </button>
                          </div>
                        )}
                        {/* Suggestions below input */}
                        {visibleExpenseSuggestions.length > 0 && (
                            <div className="mt-4 flex flex-wrap gap-2">
                               {visibleExpenseSuggestions.map((suggestion) => (
                                  <button
                                    key={`${suggestion.name}-${suggestion.category}`}
                                    type="button"
                                    onClick={() => {
                                      setFormData((prev) => ({
                                        ...prev,
                                        category: suggestion.category,
                                        subcategory: suggestion.name,
                                      }));
                                      setSubcategoryQuery(suggestion.name);
                                      setManualExpenseCategory(true);
                                      setShowCategoryPicker(false);
                                    }}
                                    className="inline-flex items-center gap-1.5 px-3 py-1.5 rounded-xl bg-gray-50 border border-gray-100 hover:bg-gray-100 transition-colors text-xs font-bold text-gray-600"
                                  >
                                    {getCategoryCartoonIcon(suggestion.category, 14)} {suggestion.name}
                                  </button>
                                ))}
                            </div>
                        )}
                      </div>
                    ) : (
                      <select value={formData.category}
                        onChange={e => {
                          setFormData(prev => ({ ...prev, category: e.target.value, subcategory: '' }));
                          setManualExpenseCategory(true);
                        }}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all appearance-none cursor-pointer">
                        {liveCategories[formData.type].map(cat => (
                          <option key={cat} value={cat}>{cat}</option>
                        ))}
                      </select>
                    )}
                  </div>
                  <div>
                    <input type="date" value={formData.date}
                      onChange={e => setFormData(prev => ({ ...prev, date: e.target.value }))}
                      className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                      required />
                  </div>
                </div>
              </div>

              {/* Additional Details Card */}
              {!isLoanExpense && (
                <div className="bg-white rounded-[2rem] p-8 shadow-sm border border-gray-100">
                  <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest mb-6">Additional Details (Optional)</h2>
                  <div className="grid grid-cols-2 gap-6">
                    <div>
                      <input type="text" value={formData.merchant}
                        onChange={e => setFormData(prev => ({ ...prev, merchant: e.target.value }))}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                        placeholder={isExpense ? 'Merchant (e.g. Amazon, Starbucks)' : 'Source / Payer'} />
                    </div>
                    <div>
                      <input type="text" value={formData.description}
                        onChange={e => setFormData(prev => ({ ...prev, description: e.target.value }))}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                        placeholder="Add a note..." />
                    </div>
                  </div>
                </div>
              )}

              {/* Loan Extra Fields */}
              {isLoanExpense && (
                <div className="bg-white rounded-[2rem] p-8 shadow-sm border border-gray-100">
                  <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest mb-6">Loan Details</h2>
                  <div className="flex gap-4 mb-6">
                    <button type="button" onClick={() => setLoanType('borrowed')}
                      className={cn('flex-1 py-4 rounded-2xl border-2 text-base font-bold transition-all flex items-center justify-center gap-2',
                        loanType === 'borrowed' ? 'border-sky-500 bg-sky-500 text-white shadow-md' : 'border-gray-100 bg-white text-gray-700 hover:border-gray-300')}>
                      <ArrowDownLeft size={20} /> Borrowed
                    </button>
                    <button type="button" onClick={() => setLoanType('lent')}
                      className={cn('flex-1 py-4 rounded-2xl border-2 text-base font-bold transition-all flex items-center justify-center gap-2',
                        loanType === 'lent' ? 'border-sky-500 bg-sky-500 text-white shadow-md' : 'border-gray-100 bg-white text-gray-700 hover:border-gray-300')}>
                      <ArrowUpRight size={20} /> Lent
                    </button>
                  </div>
                  <div className="grid grid-cols-2 gap-6">
                    <div>
                      <p className="text-[10px] font-bold uppercase tracking-widest text-gray-400 mb-2">
                        {loanType === 'borrowed' ? 'Lender' : 'Borrower'}
                      </p>
                      <input type="text" value={loanDraft.contactName}
                        onChange={e => setLoanDraft(prev => ({ ...prev, friendId: undefined, contactName: e.target.value }))}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                        placeholder={loanType === 'borrowed' ? 'Who gave you this loan?' : 'Who received this loan?'}
                        required={isLoanExpense} />
                    </div>
                    <div>
                      <p className="text-[10px] font-bold uppercase tracking-widest text-gray-400 mb-2">Due Date</p>
                      <input type="date" value={loanDraft.dueDate}
                        onChange={e => setLoanDraft(prev => ({ ...prev, dueDate: e.target.value }))}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                        required={isLoanExpense} min={formData.date} />
                    </div>
                    <div>
                      <p className="text-[10px] font-bold uppercase tracking-widest text-gray-400 mb-2">Interest Rate (% p.a.)</p>
                      <div className="flex items-center bg-gray-50 border border-gray-100 rounded-2xl px-5 py-4 focus-within:ring-2 focus-within:ring-black/5 focus-within:bg-white transition-all">
                        <input type="number" min="0" step="0.01" value={loanDraft.interestRate || ''}
                          onChange={e => setLoanDraft(prev => ({ ...prev, interestRate: parseFloat(e.target.value) || 0 }))}
                          className="flex-1 bg-transparent text-base font-semibold text-gray-900 focus:outline-none"
                          placeholder="0.00" />
                        <span className="text-gray-400 font-bold ml-2 text-base">%</span>
                      </div>
                    </div>
                    <div>
                      <p className="text-[10px] font-bold uppercase tracking-widest text-gray-400 mb-2">Notes</p>
                      <input type="text" value={loanDraft.notes}
                        onChange={e => setLoanDraft(prev => ({ ...prev, notes: e.target.value }))}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                        placeholder="Optional note..." />
                    </div>
                  </div>
                </div>
              )}

              {/* Group Extra Fields */}
              {isGroupExpense && (
                <div className="bg-white rounded-[2rem] p-8 shadow-sm border border-gray-100">
                  <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest mb-6">Group Split Details</h2>
                  <div className="grid grid-cols-2 gap-6 mb-6">
                    <div>
                      <p className="text-[10px] font-bold uppercase tracking-widest text-gray-400 mb-2">Group Name</p>
                      <input type="text" value={groupName}
                        onChange={e => setGroupName(e.target.value)}
                        className="w-full py-4 px-5 bg-gray-50 border border-gray-100 rounded-2xl text-base font-semibold text-gray-900 focus:outline-none focus:ring-2 focus:ring-black/5 focus:bg-white transition-all"
                        placeholder="Dinner, Goa trip, team lunch..."
                        required={isGroupExpense} />
                    </div>
                    <div>
                      <p className="text-[10px] font-bold uppercase tracking-widest text-gray-400 mb-2">Split Type</p>
                      <div className="flex gap-2">
                        <button type="button" onClick={() => setGroupSplitType('equal')}
                          className={cn('flex-1 py-4 rounded-2xl border-2 text-sm font-bold transition-all',
                            groupSplitType === 'equal' ? 'border-gray-900 bg-gray-900 text-white shadow-md' : 'border-gray-100 bg-white text-gray-700')}>
                          Equal Split
                        </button>
                        <button type="button" onClick={() => setGroupSplitType('custom')}
                          className={cn('flex-1 py-4 rounded-2xl border-2 text-sm font-bold transition-all',
                            groupSplitType === 'custom' ? 'border-gray-900 bg-gray-900 text-white shadow-md' : 'border-gray-100 bg-white text-gray-700')}>
                          Custom Split
                        </button>
                      </div>
                    </div>
                  </div>
                  <div className="flex gap-2 mb-4">
                    <button type="button" onClick={() => addGroupParticipant()}
                      className="inline-flex items-center gap-2 px-5 py-3 rounded-xl bg-gray-50 border border-gray-100 text-sm font-bold text-gray-700 hover:bg-gray-100 transition-colors">
                      <UserPlus size={16} /> Add Participant
                    </button>
                  </div>
                  <div className="space-y-3">
                    {groupParticipants.map((p, i) => (
                      <div key={p.id} className="flex items-center gap-4 bg-gray-50 border border-gray-100 rounded-2xl px-5 py-4">
                        <div className="flex h-10 w-10 shrink-0 items-center justify-center rounded-xl bg-white border border-gray-100 text-sm font-bold text-gray-600">{i + 1}</div>
                        <input type="text" value={p.name}
                          onChange={e => updateGroupParticipant(p.id, { name: e.target.value })}
                          className="flex-1 bg-transparent text-base font-semibold text-gray-900 focus:outline-none"
                          placeholder={`Friend ${i + 1} name`} />
                        {groupSplitType === 'custom' && (
                          <div className="flex items-center gap-2 text-base">
                            <span className="text-gray-400 text-sm">{currency}</span>
                            <input type="number" min="0" step="0.01" value={p.share || ''}
                              onChange={e => updateGroupParticipant(p.id, { share: parseFloat(e.target.value) || 0 })}
                              className="w-24 bg-transparent text-base font-semibold text-gray-900 focus:outline-none text-right"
                              placeholder="0.00" />
                          </div>
                        )}
                        {groupSplitType === 'equal' && (
                          <span className="text-base font-semibold text-gray-600 shrink-0">{formatAccountBalance(equalPerPersonShare)}</span>
                        )}
                        <button type="button" onClick={() => removeGroupParticipant(p.id)}
                          className="flex h-10 w-10 shrink-0 items-center justify-center rounded-xl bg-red-50 text-red-500 hover:bg-red-100 transition-colors">
                          <Trash2 size={16} />
                        </button>
                      </div>
                    ))}
                  </div>
                </div>
              )}
            </div>

            {/* -- SIDE COLUMN (Right) -- */}
            <div className="col-span-12 lg:col-span-4 space-y-6">

              {/* Mode Selection Card */}
              {isExpense && (
                <div className="bg-white rounded-[2rem] p-6 shadow-sm border border-gray-100">
                  <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest mb-4">Transaction Mode</h2>
                  <div className="grid grid-cols-1 gap-3">
                    {(['individual', 'group', 'loan'] as const).map(mode => (
                      <button key={mode} type="button" onClick={() => switchExpenseMode(mode)}
                        className={cn('flex items-center justify-between px-5 py-4 rounded-2xl border-2 transition-all',
                          expenseMode === mode ? 'border-gray-900 bg-gray-900 text-white shadow-md' : 'border-gray-100 bg-white text-gray-700 hover:border-gray-300')}>
                        <div className="flex items-center gap-3">
                           {mode === 'individual' && <Wallet size={18} />}
                           {mode === 'group' && <Users size={18} />}
                           {mode === 'loan' && <Banknote size={18} />}
                           <span className="font-bold capitalize text-base">{mode === 'loan' ? 'Loan / Debt' : mode}</span>
                        </div>
                        {expenseMode === mode && <Check size={18} className="text-white" />}
                      </button>
                    ))}
                  </div>
                </div>
              )}

              {/* Payment Account Card */}
              <div className="bg-white rounded-[2rem] p-6 shadow-sm border border-gray-100">
                <h2 className="text-sm font-bold text-gray-400 uppercase tracking-widest mb-4">Payment Account</h2>
                <div className="space-y-3">
                   {accounts.map(acc => {
                     const isSelected = formData.accountId === acc.id;
                     const meta = accountTypeMeta[acc.type ?? 'bank'];
                     return (
                       <button
                         key={acc.id}
                         type="button"
                         onClick={() => setFormData(prev => ({ ...prev, accountId: acc.id }))}
                         className={cn('w-full flex items-center justify-between p-4 rounded-2xl border-2 transition-all',
                            isSelected ? 'border-blue-500 bg-blue-50 shadow-sm' : 'border-gray-100 bg-white hover:border-gray-300'
                         )}>
                          <div className="flex items-center gap-3">
                             <div className={cn('flex h-10 w-10 items-center justify-center rounded-xl', meta.shell)}>
                               <meta.icon size={18} />
                             </div>
                             <div className="text-left">
                               <p className={cn('font-bold text-base leading-tight', isSelected ? 'text-blue-900' : 'text-gray-900')}>{acc.name}</p>
                               <p className={cn('text-xs font-semibold', isSelected ? 'text-blue-600' : 'text-gray-500')}>
                                 {formatAccountBalance(acc.balance)} available
                               </p>
                             </div>
                          </div>
                          {isSelected && <div className="h-5 w-5 rounded-full bg-blue-500 flex items-center justify-center"><Check size={12} className="text-white" /></div>}
                       </button>
                     );
                   })}
                </div>
              </div>

              {/* Submit Card */}
              <div className="bg-white rounded-[2rem] p-6 shadow-sm border border-gray-100 sticky top-10">
                 <div className="mb-6">
                    <p className="text-gray-500 text-sm font-bold uppercase tracking-wider mb-2">Total Amount</p>
                    <p className={cn('text-4xl font-black bg-clip-text text-transparent bg-gradient-to-r', accent.amountCard)}>
                      {currency}{formData.amount > 0 ? formData.amount.toFixed(2) : '0.00'}
                    </p>
                 </div>
                 
                 <motion.button
                  type="submit"
                  disabled={isSubmitting || !formData.amount}
                  whileTap={{ scale: 0.97 }}
                  className={cn(
                    'w-full py-5 rounded-[1.25rem] font-black text-white text-lg shadow-lg bg-gradient-to-r transition-all disabled:opacity-50 disabled:cursor-not-allowed flex items-center justify-center gap-3',
                    accent.btn
                  )}>
                  {isSubmitting
                    ? <><span className="w-5 h-5 border-2 border-white/40 border-t-white rounded-full animate-spin" /> Saving...</>
                    : <><Zap size={20} /> {isLoanExpense ? (loanType === 'borrowed' ? 'Create Borrow Record' : 'Create Lend Record') : isGroupExpense ? 'Create Group Expense' : `Add ${isExpense ? 'Expense' : 'Income'}`}</>
                  }
                </motion.button>
              </div>

            </div>
          </form>
        </div>
      </div>
